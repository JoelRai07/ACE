\section{To-Do-Liste für Kapitel 7}

\subsection*{A. Ziel und Forschungsfragen sauber verankern}
\begin{itemize}
    \item Evaluationsziel in 3--5 Sätzen formulieren: Vergleich von Detektion, Priorisierung und Robustheit.
    \item \textbf{FF1} operationalisieren: Beitrag lokaler \ac{LLM}s zur Priorisierung und Aufbereitung.
    \item \textbf{FF2} operationalisieren: Mehrwert von Codekontext und LLM-Detektor gegenüber Tool-only.
\end{itemize}

\subsection*{B. Versuchsdesign dokumentieren}
\begin{itemize}
    \item Modelle nennen und kurz begründen: Qwen2.5-coder-7b, Qwen2.5-coder-14b, Qwen3-32b, DeepSeek-r1-70b.
    \item Testobjekte beschreiben: 12-, 50- und 100-Fehler-Suite.
    \item Bedingungen fixieren:
    \begin{itemize}
        \item axe-only
        \item tools-only (axe + Playwright + grep)
        \item tools + LLM-Priorisierung
        \item tools + LLM-Detektor + LLM-Priorisierung
        \item optional: tools + LLM-Detektor + LLM-Priorisierung ohne grep
    \end{itemize}
    \item Run-Anzahl je Bedingung festhalten (z. B. 10 Runs für 12er, 5 Runs für 50/100).
    \item Reproduzierbarkeit dokumentieren: identische Prompts, Temperature, Hardware/VM, Zeitfenster.
\end{itemize}

\subsection*{C. Metriken fest definieren}
\begin{itemize}
    \item \textbf{Detektion}: Recall, Precision, F1, False-Positive-Rate.
    \item \textbf{Qualität der Empfehlung}: Stufe 0/1/2 gemäß Tab.~\ref{tab:bewertungsskala_empfehlungen}.
    \item \textbf{Formatstabilität}: Parsing-Quote (\texttt{parsed.success}).
    \item \textbf{Effizienz}: Laufzeit pro Phase, Gesamtlaufzeit, Prompt-/Output-Tokens.
\end{itemize}

\subsection*{D. Auswertung in 3 Dimensionen (Prüfer-Fokus)}
\textbf{Dimension 1: Verstoßkategorie} (syntaktisch, semantisch, layout)
\begin{itemize}
    \item Für jede Kategorie Recall/Precision pro Bedingung berichten.
    \item Kategorieabhängige Stärken/Schwächen der Modelle diskutieren.
\end{itemize}

\textbf{Dimension 2: WCAG-Kriteriumsebene} (z. B. 1.1.1, 2.1.1, 2.4.7)
\begin{itemize}
    \item Pro Kriterium Trefferquote je Modell/Bedingung darstellen.
    \item Besonders kritische Kriterien explizit vergleichen (z. B. 2.1.1 Keyboard, 2.4.7 Fokus sichtbar).
\end{itemize}

\textbf{Dimension 3: Pipeline-Bedingung / Modellgröße}
\begin{itemize}
    \item Effekt von Komponenten (Ablation) und Modellgröße auf Qualität und Laufzeit zeigen.
    \item Trade-off diskutieren: Qualität vs. Laufzeit/Tokenkosten.
\end{itemize}

\subsection*{E. Ergebnisdarstellung (quantitativ)}
\begin{itemize}
    \item Tabelle: Gesamtmetriken pro Modell \(\times\) Bedingung \(\times\) Suite.
    \item Tabelle: Metriken nach den 3 Dimensionen.
    \item Tabelle: Laufzeit und Tokens (Median, Mittelwert, Streuung).
    \item Tabellen im Anhang referenzieren und bei Bedarf ergänzen.
\end{itemize}

\subsection*{F. Ergebnisdarstellung (qualitativ)}
\begin{itemize}
    \item Mindestens drei Fallbeispiele: guter Fix, teilweise brauchbar, Halluzination.
    \item Je Fall: Input-Finding, Modellantwort, Bewertung (0/1/2), kurze Begründung.
\end{itemize}

\subsection*{G. Limitationen und Validität}
\begin{itemize}
    \item axe-core nur initialer Zustand (kein vollständiger Re-Scan je Zustand).
    \item Regex-Limits von grep und mögliche False Positives.
    \item Modellbedingte Halluzinationen und Formatabweichungen.
    \item Externe Validität: zunächst nur eine SPA-Klasse als Testobjekt.
\end{itemize}
