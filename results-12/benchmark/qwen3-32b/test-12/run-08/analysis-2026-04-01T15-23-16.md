### [llm-006] Modal ohne Fokus-Trap
- **WCAG:** 2.1.2 (Level A)
- **Schweregrad:** critical
- **Element/Datei:** components/ContactForm.tsx:102
- **Problem:** Der modale Dialog hat keinen Fokus-Trap, wodurch der Fokus in den Hintergrund wechseln kann.
- **Fix:**
```tsx
import { useEffect } from 'react';

const Modal = ({ isOpen, onClose }) => {
  useEffect(() => {
    if (isOpen) {
      const firstFocusable = document.querySelector('#modal-title');
      const lastFocusable = document.querySelector('#close-button');
      const trapFocus = (e) => {
        const isTabPressed = e.key === 'Tab';
        if (!isTabPressed) return;

        if (e.shiftKey) {
          if (document.activeElement === firstFocusable) {
            e.preventDefault();
            lastFocusable.focus();
          }
        } else {
          if (document.activeElement === lastFocusable) {
            e.preventDefault();
            firstFocusable.focus();
          }
        }
      };
      window.addEventListener('keydown', trapFocus);
      return () => window.removeEventListener('keydown', trapFocus);
    }
  }, [isOpen]);

  return (
    <div role="dialog" aria-modal="true" aria-labelledby="modal-title">
      <h2 id="modal-title">Nachricht gesendet</h2>
      <button id="close-button" onClick={onClose}>Schließen</button>
    </div>
  );
};
```
